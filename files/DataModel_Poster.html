DAX Measures Quick Reference
📊 CORE KPIs
Total Patients = COUNTROWS(Fact_Patients)
Avg Billing = AVERAGE([BillingAmountINR])
Avg LOS = AVERAGE([LengthOfStay])
Total Billing = SUM([BillingAmountINR])
⏱ TIME INTELLIGENCE
YTD Patients = TOTALYTD([Total Patients], Dim_Date[Date])
PYTD = CALCULATE([Total Patients], SAMEPERIODLASTYEAR(...))
YoY % = DIVIDE([Total Patients]-[PYTD], [PYTD], 0)
MoM % = DIVIDE(curr - prev, prev, 0)
🔢 ADMISSION SPLITS
Emergency % = DIVIDE(CALCULATE([Total Patients],
  [AdmissionType]="Emergency"), [Total Patients], 0)
Disease Prev % = DIVIDE([Total Patients],
  CALCULATE([Total Patients], ALL(Dim_Disease)), 0)
